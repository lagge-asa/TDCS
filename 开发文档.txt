2.2 定时轮询兜底（PollingScanner）
事件驱动可能因 OS 缓冲区溢出、网络文件系统等原因丢失事件，定时轮询作为兜底。

python

class PollingScanner:
    """定时全量/增量扫描，弥补事件监听遗漏"""

    def __init__(self, task_config, state_tracker, task_queue):
        self.task_config = task_config
        self.state_tracker = state_tracker
        self.task_queue = task_queue
        self._stop_event = threading.Event()

    def start(self):
        """启动轮询线程"""
        interval = self.task_config.schedule.poll_interval
        if interval <= 0:
            return  # 仅事件驱动模式
        thread = threading.Thread(target=self._poll_loop, daemon=True)
        thread.start()

    def _poll_loop(self):
        while not self._stop_event.is_set():
            self._scan_folder()
            self._stop_event.wait(self.task_config.schedule.poll_interval)

    def _scan_folder(self):
        """扫描文件夹，将未处理的文件入队"""
        folder = self.task_config.monitor.folder_path
        extensions = self.task_config.monitor.file_extensions
        for root, dirs, files in os.walk(folder):
            for fname in files:
                if any(fname.endswith(ext) for ext in extensions):
                    fpath = os.path.join(root, fname)
                    if not self.state_tracker.is_processed(fpath):
                        self.task_queue.put((self.task_config.task_id, fpath))
2.3 ETL Pipeline 编排
将 Extract → Transform → Load 统一编排，支持前置/后置钩子。

python

class ETLPipeline:
    """标准 ETL 流水线"""

    def __init__(self, extractor, transformer, table_router, loader, encryption=None):
        self.extractor = extractor
        self.transformer = transformer
        self.table_router = table_router
        self.loader = loader
        self.encryption = encryption
        self._pre_hooks = []    # 前置钩子（如：记录开始时间、初始化计数器）
        self._post_hooks = []   # 后置钩子（如：发送通知、更新质量报告）

    def add_pre_hook(self, hook):
        self._pre_hooks.append(hook)

    def add_post_hook(self, hook):
        self._post_hooks.append(hook)

    def execute(self, file_path, task_config) -> PipelineResult:
        """执行完整 ETL 流程"""
        context = PipelineContext(file_path=file_path, task_config=task_config)

        try:
            # 1. 前置钩子
            for hook in self._pre_hooks:
                hook(context)

            # 2. Extract
            rows = self.extractor.extract(file_path)
            context.raw_count = len(rows)

            # 3. Transform
            transformed = []
            for row in rows:
                result = self.transformer.transform(row)
                if result is not None:  # None 表示过滤掉
                    transformed.append(result)
            context.transformed_count = len(transformed)

            # 4. Encryption
            if self.encryption:
                transformed = self.encryption.encrypt_fields(transformed)

            # 5. Route & Load（按月表分组）
            grouped = self.table_router.group_by_table(transformed, task_config)
            for table_name, table_rows in grouped.items():
                self.table_router.ensure_table_exists(table_name, task_config)
                self.loader.load(table_name, table_rows)

            context.status = PipelineStatus.SUCCESS

        except RetryableError as e:
            context.status = PipelineStatus.RETRY
            context.error = e
        except FatalError as e:
            context.status = PipelineStatus.FAILED
            context.error = e
        except SkipFileError as e:
            context.status = PipelineStatus.SKIPPED
            context.error = e

        # 6. 后置钩子
        for hook in self._post_hooks:
            hook(context)

        return PipelineResult.from_context(context)

class ETLError(Exception):
    """ETL 基础异常"""
    pass

class RetryableError(ETLError):
    """可重试异常 — 网络超时、数据库连接断开、文件被占用等
    处理策略：按退避策略重试，超过 max_retries 后进入死信目录"""
    pass

class FatalError(ETLError):
    """致命异常 — 配置错误、SQL 语法错误、加密密钥无效等
    处理策略：不重试，立即进入死信目录，触发告警"""
    pass

class SkipFileError(ETLError):
    """跳过异常 — 文件格式不支持、空文件、文件已处理等
    处理策略：标记为 SKIPPED，不重试，不入死信目录"""
    pass

class DataQualityError(ETLError):
    """数据质量异常 — 单行数据格式错误
    处理策略：记录错误行，可选择跳过该行继续或整文件回滚"""
    pass

class DistributedLock:
    """基于 MySQL GET_LOCK() 的分布式锁"""

    def acquire(self, cursor, file_path: str, timeout: int = 5) -> bool:
        lock_name = f"etl_file_{hashlib.md5(file_path.encode()).hexdigest()}"
        cursor.execute("SELECT GET_LOCK(%s, %s)", (lock_name, timeout))
        result = cursor.fetchone()[0]
        return result == 1

    def release(self, cursor, file_path: str) -> None:
        lock_name = f"etl_file_{hashlib.md5(file_path.encode()).hexdigest()}"
        cursor.execute("SELECT RELEASE_LOCK(%s)", (lock_name,))

class StreamingExtractor:
    """大文件流式读取，避免内存溢出"""

    # 文件大小阈值（超过此值使用流式处理）
    STREAM_THRESHOLD = 100 * 1024 * 1024  # 100MB

    def extract(self, file_path: str, transformer) -> Iterator[List[dict]]:
        """按批次 yield 数据，每批 batch_size 行"""
        file_size = os.path.getsize(file_path)
        batch_size = 1000 if file_size > self.STREAM_THRESHOLD else None

        ext = Path(file_path).suffix.lower()
        if ext == ".csv":
            yield from self._stream_csv(file_path, batch_size)
        elif ext == ".json":
            yield from self._stream_json(file_path, batch_size)
        elif ext in (".xlsx", ".xls"):
            # Excel 文件不支持真正的流式读取，使用 openpyxl read-only 模式
            yield from self._stream_excel(file_path, batch_size)

    def _stream_csv(self, file_path, batch_size):
        import csv
        with open(file_path, 'r', encoding='utf-8-sig') as f:
            reader = csv.DictReader(f)
            batch = []
            for row in reader:
                batch.append(row)
                if batch_size and len(batch) >= batch_size:
                    yield batch
                    batch = []
            if batch:
                yield batch

    def _stream_json(self, file_path, batch_size):
        import ijson  # 流式 JSON 解析器
        with open(file_path, 'rb') as f:
            batch = []
            for item in ijson.items(f, 'item'):
                batch.append(item)
                if batch_size and len(batch) >= batch_size:
                    yield batch
                    batch = []
            if batch:
                yield batch

import logging

class StructuredFormatter(logging.Formatter):
    """JSON 结构化日志"""

    def format(self, record):
        log_entry = {
            "timestamp": datetime.utcnow().isoformat(),
            "level": record.levelname,
            "logger": record.name,
            "message": record.getMessage(),
            "trace_id": getattr(record, "trace_id", ""),
            "task_id": getattr(record, "task_id", ""),
            "file_path": getattr(record, "file_path", ""),
            "instance_id": getattr(record, "instance_id", ""),
        }
        if record.exc_info:
            log_entry["exception"] = self.formatException(record.exc_info)
        return json.dumps(log_entry, ensure_ascii=False)

2.14 Worker 健康监控与自动恢复

class WorkerSupervisor:
    """工作线程健康监控与自动恢复"""

    def __init__(self, worker_threads: list, check_interval=30):
        self.workers = worker_threads
        self.check_interval = check_interval
        self._stop_event = threading.Event()

    def start(self):
        thread = threading.Thread(target=self._supervise_loop, daemon=True)
        thread.start()

    def _supervise_loop(self):
        while not self._stop_event.is_set():
            for i, worker in enumerate(self.workers):
                if not worker.is_alive():
                    logger.warning(f"Worker-{i} 已死亡，正在重启...")
                    self._restart_worker(i)
            self._stop_event.wait(self.check_interval)

    def _restart_worker(self, index):
        old_worker = self.workers[index]
        new_worker = threading.Thread(
            target=old_worker._target,
            args=old_worker._args,
            kwargs=old_worker._kwargs,
            daemon=True,
            name=f"Worker-{index}"
        )
        self.workers[index] = new_worker
        new_worker.start()

CREATE DATABASE IF NOT EXISTS etl_db
    DEFAULT CHARSET utf8mb4
    DEFAULT COLLATE utf8mb4_unicode_ci;

USE etl_db;

-- 1. 文件处理状态表
CREATE TABLE IF NOT EXISTS processed_files (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    task_id VARCHAR(100) NOT NULL,
    file_path VARCHAR(500) NOT NULL,
    file_name VARCHAR(255) NOT NULL,
    file_size BIGINT,
    file_modified BIGINT NOT NULL,
    file_hash VARCHAR(64),                  -- MD5 哈希，防止重复内容
    status ENUM('PENDING','PROCESSING','SUCCESS','FAILED','SKIPPED') DEFAULT 'PENDING',
    retry_count INT DEFAULT 0,
    error_message TEXT,
    error_type VARCHAR(50),                 -- RetryableError / FatalError / SkipFileError
    row_count INT DEFAULT 0,
    valid_row_count INT DEFAULT 0,
    processing_time_ms INT,
    instance_id VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    processed_at TIMESTAMP NULL,
    UNIQUE KEY uk_task_file (task_id, file_path, file_modified),
    INDEX idx_status (status),
    INDEX idx_task_status (task_id, status),
    INDEX idx_created (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 2. Leader 选举表
CREATE TABLE IF NOT EXISTS leader (
    id INT PRIMARY KEY DEFAULT 1,
    instance_id VARCHAR(100),
    last_heartbeat TIMESTAMP NULL,
    started_at TIMESTAMP NULL,
    version INT DEFAULT 0                   -- 乐观锁版本号
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

    target VARCHAR(200),
    detail JSON,

    cpu_percent DECIMAL(5,2),
    INDEX idx_instance_time (instance_id, heartbeat_time)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

### 2.2 定时轮询兜底（PollingScanner）

> 事件驱动可能因 OS 缓冲区溢出、网络文件系统等原因丢失事件，定时轮询作为兜底。

```python
class PollingScanner:
    """定时全量/增量扫描，弥补事件监听遗漏"""

    def __init__(self, task_config, state_tracker, task_queue):
        self.task_config = task_config
        self.state_tracker = state_tracker
        self.task_queue = task_queue
        self._stop_event = threading.Event()

    def start(self):
        """启动轮询线程"""
        interval = self.task_config.schedule.poll_interval
        if interval <= 0:
            return  # 仅事件驱动模式
        thread = threading.Thread(target=self._poll_loop, daemon=True)
        thread.start()

    def _poll_loop(self):
        while not self._stop_event.is_set():
            self._scan_folder()
            self._stop_event.wait(self.task_config.schedule.poll_interval)

    def _scan_folder(self):
        """扫描文件夹，将未处理的文件入队"""
        folder = self.task_config.monitor.folder_path
        extensions = self.task_config.monitor.file_extensions
        for root, dirs, files in os.walk(folder):
            for fname in files:
                if any(fname.endswith(ext) for ext in extensions):
                    fpath = os.path.join(root, fname)
                    if not self.state_tracker.is_processed(fpath):
                        self.task_queue.put((self.task_config.task_id, fpath))
```

### 2.3 ETL Pipeline 编排

> 将 Extract → Transform → Load 统一编排，支持前置/后置钩子。

```python
class ETLPipeline:
    """标准 ETL 流水线"""

    def __init__(self, extractor, transformer, table_router, loader, encryption=None):
        self.extractor = extractor
        self.transformer = transformer
        self.table_router = table_router
        self.loader = loader
        self.encryption = encryption
        self._pre_hooks = []    # 前置钩子（如：记录开始时间、初始化计数器）
        self._post_hooks = []   # 后置钩子（如：发送通知、更新质量报告）

    def add_pre_hook(self, hook):
        self._pre_hooks.append(hook)

    def add_post_hook(self, hook):
        self._post_hooks.append(hook)

    def execute(self, file_path, task_config) -> PipelineResult:
        """执行完整 ETL 流程"""
        context = PipelineContext(file_path=file_path, task_config=task_config)

        try:
            # 1. 前置钩子
            for hook in self._pre_hooks:
                hook(context)

            # 2. Extract
            rows = self.extractor.extract(file_path)
            context.raw_count = len(rows)

            # 3. Transform
            transformed = []
            for row in rows:
                result = self.transformer.transform(row)
                if result is not None:  # None 表示过滤掉
                    transformed.append(result)
            context.transformed_count = len(transformed)

            # 4. Encryption
            if self.encryption:
                transformed = self.encryption.encrypt_fields(transformed)

            # 5. Route & Load（按月表分组）
            grouped = self.table_router.group_by_table(transformed, task_config)
            for table_name, table_rows in grouped.items():
                self.table_router.ensure_table_exists(table_name, task_config)
                self.loader.load(table_name, table_rows)

            context.status = PipelineStatus.SUCCESS

        except RetryableError as e:
            context.status = PipelineStatus.RETRY
            context.error = e
        except FatalError as e:
            context.status = PipelineStatus.FAILED
            context.error = e
        except SkipFileError as e:
            context.status = PipelineStatus.SKIPPED
            context.error = e

        # 6. 后置钩子
        for hook in self._post_hooks:
            hook(context)

        return PipelineResult.from_context(context)
```

### 2.4 异常分类体系

```python
class ETLError(Exception):
    """ETL 基础异常"""
    pass

class RetryableError(ETLError):
    """可重试异常 — 网络超时、数据库连接断开、文件被占用等
    处理策略：按退避策略重试，超过 max_retries 后进入死信目录"""
```

### 2.9 分布式锁（防重复处理）

```python
class DistributedLock:
    """基于 MySQL GET_LOCK() 的分布式锁"""

    def acquire(self, cursor, file_path: str, timeout: int = 5) -> bool:
        lock_name = f"etl_file_{hashlib.md5(file_path.encode()).hexdigest()}"
        cursor.execute("SELECT GET_LOCK(%s, %s)", (lock_name, timeout))
        result = cursor.fetchone()[0]
        return result == 1

    def release(self, cursor, file_path: str) -> None:
        lock_name = f"etl_file_{hashlib.md5(file_path.encode()).hexdigest()}"
        cursor.execute("SELECT RELEASE_LOCK(%s)", (lock_name,))
```

### 2.10 大文件流式处理

```python
class StreamingExtractor:
    """大文件流式读取，避免内存溢出"""

    # 文件大小阈值（超过此值使用流式处理）
    STREAM_THRESHOLD = 100 * 1024 * 1024  # 100MB

    def extract(self, file_path: str, transformer) -> Iterator[List[dict]]:
        """按批次 yield 数据，每批 batch_size 行"""
        file_size = os.path.getsize(file_path)
        batch_size = 1000 if file_size > self.STREAM_THRESHOLD else None

        ext = Path(file_path).suffix.lower()
        if ext == ".csv":
            yield from self._stream_csv(file_path, batch_size)
        elif ext == ".json":
            yield from self._stream_json(file_path, batch_size)
        elif ext in (".xlsx", ".xls"):
            # Excel 文件不支持真正的流式读取，使用 openpyxl read-only 模式
            yield from self._stream_excel(file_path, batch_size)

    def _stream_csv(self, file_path, batch_size):
        import csv
        with open(file_path, 'r', encoding='utf-8-sig') as f:
            reader = csv.DictReader(f)
            batch = []
            for row in reader:
                batch.append(row)
                if batch_size and len(batch) >= batch_size:
                    yield batch
                    batch = []
            if batch:
                yield batch

    def _stream_json(self, file_path, batch_size):
        import ijson  # 流式 JSON 解析器
        with open(file_path, 'rb') as f:
            batch = []
            for item in ijson.items(f, 'item'):
                batch.append(item)
                if batch_size and len(batch) >= batch_size:
                    yield batch
                    batch = []
            if batch:
                yield batch
```

### 2.14 Worker 健康监控与自动恢复

```python
class WorkerSupervisor:
    """工作线程健康监控与自动恢复"""

    def __init__(self, worker_threads: list, check_interval=30):
        self.workers = worker_threads
        self.check_interval = check_interval
        self._stop_event = threading.Event()

    def start(self):
        thread = threading.Thread(target=self._supervise_loop, daemon=True)
        thread.start()

    def _supervise_loop(self):
        while not self._stop_event.is_set():
            for i, worker in enumerate(self.workers):
                if not worker.is_alive():
                    logger.warning(f"Worker-{i} 已死亡，正在重启...")
                    self._restart_worker(i)
            self._stop_event.wait(self.check_interval)

    def _restart_worker(self, index):
        old_worker = self.workers[index]
        new_worker = threading.Thread(
            target=old_worker._target,
            args=old_worker._args,
            kwargs=old_worker._kwargs,
            daemon=True,
            name=f"Worker-{index}"
        )
        self.workers[index] = new_worker
        new_worker.start()
```

```sql
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 4. 审计日志表
CREATE TABLE IF NOT EXISTS audit_log (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    user_ip VARCHAR(45),
    user_agent VARCHAR(500),
    action VARCHAR(100) NOT NULL,
    target VARCHAR(200),
    detail JSON,
    instance_id VARCHAR(100),
    INDEX idx_time (timestamp),
    INDEX idx_action (action)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 5. 配置变更历史表
CREATE TABLE IF NOT EXISTS config_history (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    config_type ENUM('main','template','cleaner') NOT NULL,
    config_key VARCHAR(200) NOT NULL,
    old_value TEXT,
    new_value TEXT,
    changed_by VARCHAR(100),
    changed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_type_key (config_type, config_key)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 6. 汇总统计表（加速查询）
CREATE TABLE IF NOT EXISTS daily_statistics (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    task_id VARCHAR(100) NOT NULL,
    stat_date DATE NOT NULL,
    table_name VARCHAR(100),
    total_files INT DEFAULT 0,
    success_files INT DEFAULT 0,
    failed_files INT DEFAULT 0,
    total_rows INT DEFAULT 0,
    valid_rows INT DEFAULT 0,
    avg_quality_score DECIMAL(5,2),
    avg_processing_time_ms INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UNIQUE KEY uk_task_date (task_id, stat_date, table_name),
    INDEX idx_date (stat_date)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 7. 服务心跳历史表（可选，用于监控页面展示）
CREATE TABLE IF NOT EXISTS heartbeat_history (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    instance_id VARCHAR(100) NOT NULL,
    heartbeat_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    status ENUM('ACTIVE','STANDBY','STARTING','STOPPING'),
    queue_size INT DEFAULT 0,
    active_workers INT DEFAULT 0,
    memory_usage_mb DECIMAL(10,2),
    cpu_percent DECIMAL(5,2),
    INDEX idx_instance_time (instance_id, heartbeat_time)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

### 6.3 第一阶段详细任务：核心引擎

**目标**：实现端到端的单文件 ETL 流程 — 监控文件夹 → 读取文件 → 清洗 → 分表写入数据库。

| 任务 | 产出 | 验收标准 |
|------|------|----------|
| 项目搭建 | 目录结构 + `pyproject.toml` + `requirements.txt` | `pip install -e .` 成功 |
| 配置管理 | `ConfigManager` + `ConfigValidator` | 能正确加载 YAML 配置并校验 |
| 文件监控 | `DebouncedHandler` + `FileStabilityChecker` | 放入文件后 3 秒内入队 |
| 文件解析 | `Extractor` (CSV/JSON/XLSX) | 三种格式均正确解析为字典列表 |
| 清洗代码 | `Transformer` + 动态加载 | 修改 .py 文件后 10 秒内生效 |
| 分表路由 | `TableRouter` + SQL 模板 | 自动创建 `order_data_202601` 格式表 |
| 数据加载 | `Loader` + 事务 | 批量插入成功，失败回滚 |
| Pipeline | `ETLPipeline` 编排 | 完整流程跑通，有结构化日志 |

### 6.4 第二阶段详细任务：基础设施

| 任务 | 产出 | 验收标准 |
|------|------|----------|
| 状态追踪 | `StateTracker` + MySQL 存储 | 文件重启后不重复处理 |
| 分布式锁 | `DistributedLock` (MySQL GET_LOCK) | 多实例不重复处理同一文件 |
| 工作线程池 | `WorkerPool` + `WorkerSupervisor` | 线程异常死亡后自动重启 |
| 加密 | `Encryption` (Fernet/AES-GCM) | 加密后可正确解密 |
| 缓存 | `CacheManager` (Redis + Local) | 缓存命中率 > 80% |
| 轮询兜底 | `PollingScanner` | 事件遗漏的文件在轮询周期内被发现 |
| 文件归档 | `FileArchiver` (move/delete/keep) | 三种模式均正常工作 |
| 质量报告 | `DataQualityReporter` | 每批次产出质量评分 |
| 日志追踪 | `StructuredFormatter` + `TraceContext` | 全链路 trace_id 可追踪 |

### 10.2 GitHub Actions 配置要点

```yaml
name: CI/CD Pipeline
on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [develop]
  tags: ['v*']

jobs:
  lint:
    runs-on: windows-latest
    steps:
      - uses: actions/checkout@v4
      - run: pip install flake8 black mypy
      - run: flake8 src/ tests/
      - run: black --check src/ tests/
      - run: mypy src/

  test:
    runs-on: windows-latest
    services:
      mysql:
        image: mysql:8.0
        env:
          MYSQL_ROOT_PASSWORD: test
          MYSQL_DATABASE: etl_test
      redis:
        image: redis:7
    steps:
      - uses: actions/checkout@v4
      - run: pip install -r requirements.txt -r requirements-dev.txt
      - run: pytest tests/ --cov=src --cov-report=xml --cov-fail-under=80

  build:
    needs: [lint, test]
    runs-on: windows-latest
    if: startsWith(github.ref, 'refs/tags/v')
    steps:
      - run: pyinstaller installer/etl_service.spec
      - run: iscc installer/setup.iss
      - uses: actions/upload-artifact@v4
        with:
          name: ETL_Service_Setup
          path: installer/Output/ETL_Service_Setup.exe

  release:
    needs: build
    steps:
      - uses: softprops/action-gh-release@v1
        with:
          files: installer/Output/ETL_Service_Setup.exe
```

## 第十二部分：风险与应对

| # | 风险 | 概率 | 影响 | 应对措施 |
|---|------|------|------|----------|
| 1 | watchdog 在网络盘/某些文件系统上不触发事件 | 中 | 高 | 定时轮询兜底已设计，PollingScanner 可覆盖 |
| 2 | 清洗代码热更新引入 bug 导致数据处理失败 | 高 | 高 | 清洗代码有语法检查入口，失败不替换旧版本 |
| 3 | MySQL 连接池耗尽 | 中 | 高 | 连接池监控 + Prometheus 告警 + 慢查询日志 |
| 4 | 大文件内存溢出 | 低 | 高 | 流式处理 + 100MB 阈值自动切换 + 内存监控 |
| 5 | Leader 选举脑裂 | 低 | 极高 | MySQL 单点写 + 乐观锁版本号 + 心跳超时双重检查 |
| 6 | PyInstaller 打包后运行异常 | 中 | 中 | 打包前完整测试 + CI 自动构建验证 |
| 7 | Windows 服务权限不足 | 中 | 中 | 安装包以管理员运行 + 配置文件权限检查 |
| 8 | 加密密钥泄露 | 低 | 极高 | 密钥仅存环境变量 + 不记日志 + 审计追踪 |
